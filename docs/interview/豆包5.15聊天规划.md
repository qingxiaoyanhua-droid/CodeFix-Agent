import re
from typing import List, Dict, Optional, Tuple
from tree_sitter import Language, Parser, Node
from tree_sitter_languages import get_language, get_parser

class ASTCodeProcessor:
    """
    基于Tree-sitter的代码AST处理引擎
    核心功能：函数定位、函数替换、导入修复、依赖分析
    完全解决行号偏移问题，支持Python/Java/JavaScript
    """
    
    SUPPORTED_LANGUAGES = {
        "python": {"ext": ".py", "function_node": "function_definition"},
        "java": {"ext": ".java", "function_node": "method_declaration"},
        "javascript": {"ext": ".js", "function_node": "function_declaration"}
    }
    
    def __init__(self, language: str = "python"):
        if language not in self.SUPPORTED_LANGUAGES:
            raise ValueError(f"不支持的语言: {language}，支持: {list(self.SUPPORTED_LANGUAGES.keys())}")
        
        self.language = language
        self.parser = get_parser(language)
        self.language_obj = get_language(language)
        self.config = self.SUPPORTED_LANGUAGES[language]
        
        # 预编译查询语句（性能优化）
        self.function_query = self.language_obj.query(f"""
            ({self.config['function_node']}) @function
        """)
        
        self.import_query = self.language_obj.query("""
            (import_statement) @import
            (import_from_statement) @import_from
        """)
        
        self.class_query = self.language_obj.query("""
            (class_definition) @class
        """)
    
    def parse_code(self, code: str) -> Node:
        """解析代码生成AST根节点"""
        return self.parser.parse(code.encode("utf-8")).root_node
    
    def get_all_functions(self, code: str) -> List[Dict]:
        """
        获取文件中所有函数的信息
        返回: [{"name": "函数名", "start_byte": 起始字节, "end_byte": 结束字节, 
               "start_point": (行, 列), "end_point": (行, 列), "node": AST节点}]
        """
        tree = self.parse_code(code)
        functions = []
        
        captures = self.function_query.captures(tree)
        for node, _ in captures:
            # 提取函数名
            name_node = node.child_by_field_name("name")
            if not name_node:
                continue
                
            function_name = name_node.text.decode("utf-8")
            functions.append({
                "name": function_name,
                "start_byte": node.start_byte,
                "end_byte": node.end_byte,
                "start_point": node.start_point,
                "end_point": node.end_point,
                "node": node
            })
        
        return functions
    
    def find_function_by_name(self, code: str, function_name: str) -> Optional[Dict]:
        """精确查找指定名称的函数"""
        functions = self.get_all_functions(code)
        for func in functions:
            if func["name"] == function_name:
                return func
        return None
    
    def find_function_by_line_number(self, code: str, line_number: int) -> Optional[Dict]:
        """
        根据行号查找包含该行的函数
        解决编译器错误只给出行号的问题
        """
        functions = self.get_all_functions(code)
        for func in functions:
            start_line = func["start_point"][0] + 1  # Tree-sitter行号从0开始
            end_line = func["end_point"][0] + 1
            if start_line <= line_number <= end_line:
                return func
        return None
    
    def replace_function(self, original_code: str, function_name: str, new_function_code: str) -> str:
        """
        精确替换指定函数，完全解决行号偏移问题
        核心方法：按字节偏移替换，而不是按行号替换
        """
        func_info = self.find_function_by_name(original_code, function_name)
        if not func_info:
            raise ValueError(f"函数 {function_name} 未找到")
        
        # 按字节偏移替换（最精确的方式，不受行号、空格、注释影响）
        code_bytes = original_code.encode("utf-8")
        new_code_bytes = (
            code_bytes[:func_info["start_byte"]] +
            new_function_code.encode("utf-8") +
            code_bytes[func_info["end_byte"]:]
        )
        
        return new_code_bytes.decode("utf-8")
    
    def fix_imports(self, code: str, missing_imports: List[str]) -> str:
        """
        自动修复缺失的导入语句
        将新导入添加到文件顶部，避免重复导入
        """
        tree = self.parse_code(code)
        imports = self.import_query.captures(tree)
        
        if not imports:
            # 没有导入语句，直接添加到文件开头
            return "\n".join(missing_imports) + "\n\n" + code
        
        # 找到最后一个导入语句的位置
        last_import_end = max(node.end_byte for node, _ in imports)
        
        # 检查是否已经存在这些导入
        existing_imports = set()
        for node, _ in imports:
            existing_imports.add(node.text.decode("utf-8").strip())
        
        new_imports = [imp for imp in missing_imports if imp.strip() not in existing_imports]
        
        if not new_imports:
            return code
        
        # 在最后一个导入语句之后添加新导入
        code_bytes = code.encode("utf-8")
        new_code_bytes = (
            code_bytes[:last_import_end] +
            b"\n" + "\n".join(new_imports).encode("utf-8") +
            code_bytes[last_import_end:]
        )
        
        return new_code_bytes.decode("utf-8")
    
    def get_function_dependencies(self, code: str, function_name: str) -> Dict[str, List[str]]:
        """
        分析函数的依赖关系
        返回: {"calls": ["被调用的函数名"], "imports": ["使用的导入"], "variables": ["使用的全局变量"]}
        """
        func_info = self.find_function_by_name(code, function_name)
        if not func_info:
            return {"calls": [], "imports": [], "variables": []}
        
        func_node = func_info["node"]
        dependencies = {"calls": [], "imports": [], "variables": []}
        
        # 递归遍历函数体，查找函数调用
        def traverse(node: Node):
            if node.type == "call":
                func_name_node = node.child_by_field_name("function")
                if func_name_node and func_name_node.type == "identifier":
                    dependencies["calls"].append(func_name_node.text.decode("utf-8"))
            
            for child in node.children:
                traverse(child)
        
        traverse(func_node)
        
        # 去重
        dependencies["calls"] = list(set(dependencies["calls"]))
        return dependencies

# 扩展：Java语言专用处理器（继承基础类）
class JavaASTProcessor(ASTCodeProcessor):

            (method_declaration) @method

        """)

class CompileFirstReActEngine:
    def __init__(self):

        self.ast_processors = {
            "python": ASTCodeProcessor("python"),
            "java": JavaASTProcessor(),
            "javascript": JavaScriptASTProcessor()
        }
    
    def _get_ast_processor(self, file_path: str) -> ASTCodeProcessor:
        """根据文件扩展名获取对应的AST处理器"""
        if file_path.endswith(".py"):
            return self.ast_processors["python"]
        elif file_path.endswith(".java"):
            return self.ast_processors["java"]
        elif file_path.endswith(".js"):
            return self.ast_processors["javascript"]
        else:
            raise ValueError(f"不支持的文件类型: {file_path}")
    
    def fix_bug(self, file_path: str, bug_description: str, error_line: int = None) -> Tuple[str, bool]:
        """
        修复bug的主流程（修改后的版本）
        完全替换原来的行号替换逻辑
        """
        # 1. 读取原始代码
        with open(file_path, "r", encoding="utf-8") as f:
            original_code = f.read()
        
        # 2. 获取AST处理器
        processor = self._get_ast_processor(file_path)
        
        # 3. 定位bug所在的函数（核心改进）
        if error_line:
            # 从编译器错误中获取行号，精确查找函数
            buggy_func = processor.find_function_by_line_number(original_code, error_line)
        else:
            # 从bug描述中提取函数名
            # 你可以用你的小模型来提取函数名
            function_name = self.small_model.extract_function_name(bug_description)
            buggy_func = processor.find_function_by_name(original_code, function_name)
        
        if not buggy_func:
            # 无法定位函数，升级到大模型
            return self._upgrade_to_large_model(file_path, bug_description), False
        
        # 4. 提取函数代码和上下文
        buggy_func_code = original_code[buggy_func["start_byte"]:buggy_func["end_byte"]]
        
        # 5. 分析函数依赖（用于上下文扩展）
        dependencies = processor.get_function_dependencies(original_code, buggy_func["name"])
        
        # 6. 小模型生成修复代码
        fixed_func_code = self.small_model.generate_fix(
            bug_description=bug_description,
            buggy_code=buggy_func_code,
            dependencies=dependencies
        )
        
        # 7. 精确替换函数（核心改进：解决行号偏移）
        try:
            fixed_code = processor.replace_function(original_code, buggy_func["name"], fixed_func_code)
        except ValueError:
            # 替换失败，升级到大模型
            return self._upgrade_to_large_model(file_path, bug_description), False
        
        # 8. 自动修复缺失的导入
        missing_imports = self.small_model.extract_missing_imports(fixed_func_code)
        if missing_imports:
            fixed_code = processor.fix_imports(fixed_code, missing_imports)
        
        # 9. 编译验证（你现有的逻辑）
        if self._compile_and_test(file_path, fixed_code):
            return fixed_code, True
        else:
            # 编译失败，升级到大模型
            return self._upgrade_to_large_model(file_path, bug_description), False
